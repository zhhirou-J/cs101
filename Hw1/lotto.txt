#include <stdio.h>
#include <stdlib.h>
#include <time.h>

FILE* lotto;
int num[6] = {0};
void lottory(){
    int arr,c,n=0;
    for(int i=1;i<6;i++){
        num[i]=0;
    }
    while(n < 5){
        arr=rand()%69+1;
        c = 0;
        for(int i =1;i<=n;i++){
            if(num[i] == arr){
                c = 1;
            }
        }
        if(c == 0){
            num[n] = arr;
            n++;
        }
    }
    num[5] = rand()%10+1;
    for(int i=0;i<6;i++){
        fprintf(lotto,"%02d ",num[i]);
    }
}
int main(){
	lotto=fopen("lotto.txt","w+");
    int n=2;
    srand((unsigned) time(NULL));

    time(&curtime);
    fprintf(lotto,"======== lotto649 ========\n%s",ctime(&curtime));
    for(int i=0;i<n;i++){
        fprintf(lotto,"[%d] : ",i+1);
        lottory();
		fprintf(lotto,"\n");
    }
	for (int h=0;h<(5-n);h++){
        fprintf(lotto,"[%d] : ",n+h+1);
